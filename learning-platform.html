2. Query Builder Pattern

// Base Model with query builder functionality class BaseModel { static tableName = ''; static async executeQuery(query: string, params: any[] = []) { // Database connection and query execution return new Promise((resolve, reject) => { db.query(query, params, (err, results) => { if (err) reject(err); else resolve(results); }); }); } static async findById(id: number) { const query = `SELECT * FROM ${this.tableName} WHERE id = ?`; const results = await this.executeQuery(query, [id]); return results[0] || null; } static async findAll(filters = {}, orderBy = 'id', order = 'ASC') { let query = `SELECT * FROM ${this.tableName} WHERE 1=1`; const params = []; // Dynamic filter building Object.entries(filters).forEach(([key, value]) => { if (value !== undefined && value !== '') { query += ` AND ${key} = ?`; params.push(value); } }); query += ` ORDER BY ${orderBy} ${order}`; return await this.executeQuery(query, params); } static async create(data: any) { const keys = Object.keys(data); const values = Object.values(data); const placeholders = keys.map(() => '?').join(', '); const query = ` INSERT INTO ${this.tableName} (${keys.join(', ')}) VALUES (${placeholders}) `; const result = await this.executeQuery(query, values); return { id: result.insertId, ...data }; } static async update(id: number, data: any) { const keys = Object.keys(data); const values = Object.values(data); const setClause = keys.map(key => `${key} = ?`).join(', '); const query = ` UPDATE ${this.tableName} SET ${setClause} WHERE id = ? `; await this.executeQuery(query, [...values, id]); return await this.findById(id); } static async delete(id: number) { const query = `DELETE FROM ${this.tableName} WHERE id = ?`; return await this.executeQuery(query, [id]); } }

🔐 Security Implementation

1. Password Security

import bcrypt from 'bcrypt'; class SecurityService { // Hash password before storing static async hashPassword(password: string): Promise { const saltRounds = 12; // Higher = more secure but slower return await bcrypt.hash(password, saltRounds); } // Verify password during login static async verifyPassword(password: string, hash: string): Promise { return await bcrypt.compare(password, hash); } // Password strength validation static validatePassword(password: string): { isValid: boolean; errors: string[] } { const errors = []; if (password.length < 8) { errors.push('Password must be at least 8 characters long'); } if (!/[A-Z]/.test(password)) { errors.push('Password must contain at least one uppercase letter'); } if (!/[a-z]/.test(password)) { errors.push('Password must contain at least one lowercase letter'); } if (!/\d/.test(password)) { errors.push('Password must contain at least one number'); } if (!/[!@#$%^&*(),.?":{}|<>]/.test(password)) { errors.push('Password must contain at least one special character'); } return { isValid: errors.length === 0, errors }; } }

2. JWT Authentication

import jwt from 'jsonwebtoken'; class AuthService { // Generate JWT token static generateToken(payload: any): string { return jwt.sign( payload, process.env.JWT_SECRET!, { expiresIn: '24h', issuer: 'movie-watchlist', audience: 'movie-users' } ); } // Verify JWT token static verifyToken(token: string): any { try { return jwt.verify(token, process.env.JWT_SECRET!); } catch (error) { if (error instanceof jwt.TokenExpiredError) { throw new Error('Token has expired'); } else if (error instanceof jwt.JsonWebTokenError) { throw new Error('Invalid token'); } else { throw new Error('Token verification failed'); } } } // Refresh token mechanism static generateRefreshToken(userId: number): string { return jwt.sign( { userId, type: 'refresh' }, process.env.REFRESH_TOKEN_SECRET!, { expiresIn: '7d' } ); } static async refreshAccessToken(refreshToken: string): Promise { const payload = jwt.verify(refreshToken, process.env.REFRESH_TOKEN_SECRET!); if (payload.type !== 'refresh') { throw new Error('Invalid refresh token'); } // Generate new access token return this.generateToken({ userId: payload.userId, type: 'access' }); } }

import validator from 'validator'; class ValidationService { // Email validation static validateEmail(email: string): boolean { return validator.isEmail(email) && email.length <= 100; } // Sanitize HTML input to prevent XSS static sanitizeInput(input: string): string { return validator.escape(input.trim()); } // Movie data validation static validateMovieData(data: any): { isValid: boolean; errors: string[] } { const errors = []; // Title validation if (!data.title || typeof data.title !== 'string') { errors.push('Title is required and must be a string'); } else if (data.title.length > 255) { errors.push('Title must be less than 255 characters'); } // Genre validation if (!Array.isArray(data.genre) || data.genre.length === 0) { errors.push('At least one genre must be specified'); } // Release date validation if (data.release_date && !validator.isDate(data.release_date)) { errors.push('Invalid release date format'); } return { isValid: errors.length === 0, errors }; } // File upload validation static validateImageUpload(file: any): { isValid: boolean; error?: string } { // Check file size (5MB limit) if (file.size > 5 * 1024 * 1024) { return { isValid: false, error: 'File size must be less than 5MB' }; } // Check file type const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp']; if (!allowedTypes.includes(file.mimetype)) { return { isValid: false, error: 'Only image files are allowed' }; } return { isValid: true }; } }

📝 MCQ Quiz - Test Your Knowledge

💼 Interview Questions & Answers

1. Explain the MVC architecture pattern used in your project.

Answer: MVC (Model-View-Controller) is an architectural pattern that separates application logic into three interconnected components:

  • Model: Handles data logic and database operations (e.g., MovieModel, UserModel)
  • View: Presents data to users (React components)
  • Controller: Handles user input and coordinates between Model and View (e.g., MovieController)

This separation improves code maintainability, testability, and allows multiple developers to work on different parts simultaneously.

2. How do you handle authentication and authorization in your application?

Answer: I implement a multi-layered security approach:

  • Authentication: JWT tokens with email verification
  • Password Security: bcrypt hashing with salt rounds
  • Authorization: Role-based access control (user/admin)
  • Token Management: Access tokens (24h) and refresh tokens (7d)
  • Middleware: Protect routes with authentication middleware

3. What are React Hooks and how do you use them in your project?

Answer: React Hooks are functions that let you use state and lifecycle features in functional components:

  • useState: Manage component state (form data, loading states)
  • useEffect: Handle side effects (API calls, subscriptions)
  • useContext: Access context values (AuthContext)
  • useNavigate: Programmatic navigation
  • Custom Hooks: useAuth for authentication logic

4. How do you optimize database queries in your application?

Answer: I use several optimization techniques:

  • Indexing: Create indexes on frequently queried columns
  • Query Builder: Dynamic query building to avoid unnecessary WHERE clauses
  • Pagination: LIMIT and OFFSET to prevent large result sets
  • Connection Pooling: Reuse database connections
  • Prepared Statements: Prevent SQL injection and improve performance

5. Explain error handling strategy in your full-stack application.

Answer: I implement comprehensive error handling:

  • Backend: Try-catch blocks, error middleware, HTTP status codes
  • Frontend: Error boundaries, loading states, user-friendly messages
  • Validation: Input validation on both client and server
  • Logging: Console logs and potential external logging service
  • Graceful Degradation: App continues functioning with partial failures

6. How would you scale this application for production?

Answer: Scaling strategies I would implement:

  • Database: Read replicas, connection pooling, query optimization
  • Caching: Redis for session storage and frequently accessed data
  • CDN: Serve static assets (images, CSS, JS) from CDN
  • Load Balancing: Multiple server instances behind load balancer
  • Monitoring: Application performance monitoring and logging
  • Security: Rate limiting, HTTPS, security headers

7. What testing strategies would you implement?

Answer: Comprehensive testing approach:

  • Unit Tests: Jest for individual functions and components
  • Integration Tests: Test API endpoints and database operations
  • Component Tests: React Testing Library for UI components
  • E2E Tests: Cypress or Playwright for user workflows
  • API Tests: Postman or Supertest for endpoint testing

8. How do you handle file uploads securely?

Answer: Secure file upload implementation:

  • Validation: File type, size, and content validation
  • Storage: Store files outside web root
  • Naming: Generate unique filenames to prevent conflicts
  • Scanning: Virus scanning for uploaded files
  • Limits: Size and rate limiting
  • CDN/Cloud: Use AWS S3 or similar for production
